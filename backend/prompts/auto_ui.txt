import allure
import pytest
from playwright.sync_api import Page, expect
@allure.feature("Калькулятор цен Cloud.ru")
@allure.story("Начальная страница")
@allure.title("Открытие калькулятора и проверка отображения нулевой стоимости")
def test_calculator_opens_with_zero_price(page: Page):
    page.goto("https://cloud.ru/price-calculator")
    expect(page.locator("text=0 ₽ в месяц")).to_be_visible()
    page.screenshot(path="screenshots/zero_price.png")
    allure.attach.file("screenshots/zero_price.png", name="Нулевая стоимость", attachment_type=allure.attachment_type.PNG)
